QR code genertaor (python):

import qrcode
from PIL import Image

# 製造 QR Code
data = "https://www.pixiv.net/users/26058726"
qr = qrcode.QRCode(version=1, box_size=10, border=5)
qr.add_data(data)
qr.make(fit=True)
qr_img = qr.make_image(fill_color="black", back_color="white")

# 載入圖示
icon_path = "pixiv.png"
icon = Image.open(icon_path)

# 調整圖示大小
icon_size = (qr_img.size[0] // 4, qr_img.size[1] // 4)
icon.thumbnail(icon_size)

# 在 QR Code 上加上圖示
pos = ((qr_img.size[0] - icon.size[0]) // 2, (qr_img.size[1] - icon.size[1]) // 2)
qr_with_icon = qr_img.copy()
qr_with_icon.paste(icon, pos)

# 儲存 QR Code 到電腦中
qr_with_icon.save("qrcode_with_icon.png")
